```python
def generate_candidate_strategies(
    goal: str,
    count: int = 4,
    use_llm: bool = True,
) -> dict[str, Any]:
    """Generate 3-5 executable candidate strategies for one research goal."""
    # 澄清检查
    clarify = clarify_strategy_goal(text)
    
    # 生成多个候选
    for idx in range(count):
        # 尝试使用 DeepSeek
        try:
            if use_llm and is_llm_intent_parser_enabled():
                spec = parse_intent_with_deepseek(prompt)
            else:
                raise LLMIntentParserUnavailable()
        except:
            # 降级到规则引擎
            base = parse_intent_to_strategy_spec(text)
            spec = _apply_variant(base, idx, text)
        
        # 验证
        validation = validate_strategy_spec(spec)
        if validation["valid"]:
            strategies.append(validation["normalized_spec"])
    
    return {"strategies": strategies, "clarification": clarify, ...}
```

```python
def batch_backtest_strategies(
    specs: list[StrategySpec | dict[str, Any]],
    objective: str = "out_sample_sharpe",
    save_recommended: bool = False,
    top_n: int = 2,
) -> dict[str, Any]:
    """Backtest many candidate specs and return a ranked shortlist."""
    rows = []
    for index, raw in enumerate(specs):
        # 验证
        validation = validate_strategy_spec(raw)
        if not validation["valid"]:
            rows.append({"status": "invalid", "errors": ...})
            continue
        
        # 选股
        screening = run_intelligent_screening(spec)
        
        # 回测
        backtest = run_in_sample_out_sample_backtest(spec, candidates)
        
        # 评分
        score = _research_score(metrics, data_quality, overfit_flag)
        
        # 筛选
        qualified = (
            not overfit_flag
            and annual_return > 0
            and sharpe > 0
            and rating in {"A", "B", "C"}
        )
        
        rows.append({
            "status": "qualified" if qualified else "filtered",
            "score": score,
            ...
        })
    
    # 排序
    ranked = sorted(rows, key=lambda x: x["score"], reverse=True)
    
    # 可选：保存推荐策略
    if save_recommended:
        for item in ranked[:top_n]:
            if item["status"] == "qualified":
                save_strategy(item["strategy_spec"])
    
    return {"ranked_strategies": ranked, ...}
```

```python
async def rollback_strategy_version(strategy_id: str, version_id: str):
    # 1. 备份当前版本
    current = get_strategy(strategy_id)
    backup_version = save_strategy_version(
        strategy_id=strategy_id,
        spec=current["spec"],
        change_note=f"rollback backup before restoring {version_id}",
        generated_by="rollback_backup"
    )
    
    # 2. 恢复目标版本
    target_version = get_strategy_version(strategy_id, version_id)
    update_strategy(strategy_id, spec=target_version["spec"])
    
    # 3. 记录回滚
    new_version = save_strategy_version(
        strategy_id=strategy_id,
        spec=target_version["spec"],
        change_note=f"rollback to v{version_id}",
        generated_by="rollback"
    )
    
    return {"success": True, "new_version": new_version, ...}
```
